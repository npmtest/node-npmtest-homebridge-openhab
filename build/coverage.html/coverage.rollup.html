        self.log("OpenHAB WS - message from " +self.itemName+": "+ message);

    });
 
    this.ws.on('close', function close() {
        self.log("OpenHAB WS - closed connection for "+self.itemName);
        self.ws = undefined;
        self = undefined;
    });
};

            self.ws.ping();
        } else {
            clearInterval(intervalId);
        }

};
